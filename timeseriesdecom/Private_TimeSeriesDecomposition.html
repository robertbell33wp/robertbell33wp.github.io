def sqErr(row):
    return (row['Private Construction'] - row['MovAvg'])**2
    
df2['MovAvg'] = np.NaN
df2['sqErr'] = np.NaN
# Chaging the DataFrame index to DatetimeIndex data type is required for using one of the functions below
df2.index = pd.DatetimeIndex(freq='m', start=pd.Timestamp(year=2002, month=1, day=31), periods = len(df2['Private Construction']))

window = 12
window = window - window % 2
# Compute the moving average in the loop below using a window centered on the data point whose average is eing computed
for i in range(int(window/2),df2.shape[0]-int(window/2)):
    df2.loc[df2.index[i],'MovAvg'] = (0.5*df2.iloc[i - int(window/2)]['Private Construction'] + df2.iloc[i - int(window/2)+1:i + int(window/2)]['Private Construction'].sum() + 0.5*df2.iloc[i + int(window/2)]['Private Construction'])/float(window)

df2['sqErr'] = (df2['Private Construction'] - df2['MovAvg'])**2
# The squared error can eb computed also with the dfHA.apply() method below
#dfHS['sqErr'] = dfHS.apply(sqErr,axis='columns')

# The moving average cannot be applied to all rows and we need to delete those rows because we cannot use them in the analysis
df2.dropna(how='any',inplace=True)

fig,ax = plt.subplots()
ax.plot(df2['MovAvg'],label='Moving Avg.')
ax.plot(df2['Private Construction'],label='Construction Spending')
ax.set_xlabel('Year')
ax.set_ylabel('Construction Spednding (in millions of $)')
ax.spines['right'].set_visible(False)
ax.spines['top'].set_visible(False)
print('Average Squared Error per Month: ',sum(df2['sqErr'])/len(df2))
print(df2)
